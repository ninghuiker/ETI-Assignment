pipeline {
    agent { dockerfile true }

    stages {
        stage('Build') {
            steps {
                sh 'python -m compileall .'
            }
        }

        stage('Test') {
            steps {
                sh 'pytest --cov'
            }

            post {
                always {
                    junit(keepLongStdio: true, allowEmptyResults: true, testResults: 'target/surefire-reports/TEST-*.xml')
                }
            }
        }
    }
}
