pipeline {
    agent { Dockerfile true }

    stages {
        stage('Build') {
            steps {
                sh 'python -m compileall .'
            }
        }

        stage('Test') {
            steps {
                sh 'pytest  --cov --cov-fail-under 35 --junitxml=coverage.xml'
            }

            post {
                always {
                    junit 'allowEmptyResults: true, coverage.xml'
                }
            }
        }
    }
}
